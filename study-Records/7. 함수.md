### break;    : 반복문이나 switch문에서 종료시킬때 사용
### return;   : 함수에서의 break처럼 사용 가능


```js
    function stopLoop(){
      while(true){
        var prom = prompt(`명령어를 입력하세요:`);
        if(prom === `멈춰`){
          break;
        }
        if(prom ===`stop`){
          return;
        }
        alert(prom);
      }//end while
      alert(`수고수고`);
    }//end function
```

### break; 는 함수 while문 하나 밖으로 나감
### return;은 while 나가고 function 까지 나감.
